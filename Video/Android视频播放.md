~~~
implementation 'com.dou361.ijkplayer:jjdxm-ijkplayer:1.0.5'
PlayerView play;
String url = "https://vip.94kuyun.com/share/PcnG4qsmeZKV1ikR";
play = new PlayerView(this)
        .setTitle("什么")
        .setScaleType(PlayStateParams.fitparent)
        .hideMenu(true)
        .forbidTouch(false)
        .setPlaySource(url);
play.startPlay();

<LinearLayout
